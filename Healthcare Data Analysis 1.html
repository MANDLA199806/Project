import pandas as pd
import numpy as np
import random
 
# Generate sample data for healthcare data analysis
np.random.seed(0)
 
# Number of patients
n = 500
 
# Generate random data
patient_ids = list(range(1, n + 1))
ages = np.random.randint(18, 80, n)
genders = np.random.choice(['Male', 'Female'], n)
diagnoses = np.random.choice(['Diabetes', 'Hypertension', 'Cancer', 'Injury'], n)
treatment_types = np.random.choice(['Surgery', 'Medication', 'Therapy'], n)
treatment_costs = np.round(np.random.uniform(1000, 50000, n), 2)
 
# Create DataFrame
data = {
    'PatientID': patient_ids,
    'Age': ages,
    'Gender': genders,
    'Diagnosis': diagnoses,
    'TreatmentType': treatment_types,
    'TreatmentCost': treatment_costs
}
 
df_healthcare = pd.DataFrame(data)
 
# Display the first 5 rows
print("Sample of Healthcare Data:")
print(df_healthcare.head())

5. Perform a statistical test to determine if there is a significant difference in treatment costs between genders¶

In [26]:
male_costs = df_healthcare[df_healthcare['Gender'] == 'Male']['TreatmentCost']
female_costs = df_healthcare[df_healthcare['Gender'] == 'Female']['TreatmentCost']
t_stat, p_value = stats.ttest_ind(male_costs, female_costs)
print(f"T-statistic: {t_stat:.2f}, P-value: {p_value:.2f}")

T-statistic: 0.48, P-value: 0.63
